        Node cur = this;

            next = cur.next;
            if (next == null) break;
            cur = next;
        }
        cur.next = node;
        return this;
    }

    public Node next() {
        return this.next;
    }

    public int getData() {
        return this.data;
    }

    public boolean isLast() {
        return this.next == null;
    }

    public void removeNext() {
        this.next = this.next.next;
    }

    public void show() {
        Node cur = this;
        while(true) {
            System.out.print(cur.data + " ");
            cur = cur.next;
            if (cur == null) break;
        }
    }

    // 后一位插入
    public void after(Node node) {
        Node nn = this.next;
        node.next = nn;
        this.next = node;
    }

}

package com.ocean.test;

public class TreeNode {

    int value;

    private TreeNode left;

    TreeNode right;

    public TreeNode(int value) {
        this.value = value;
    }

    public void setLeft(TreeNode left) {
        this.left = left;
    }

    public void setRight(TreeNode right) {
        this.right = right;
    }

    // 遍历
    // 前（中左右），中（左中右），后（左右中）
    public void frontShow() {
        System.out.println(this.value);
        if (this.left != null) this.left.frontShow();
        if (this.right != null) this.right.frontShow();
    }

    // 查找
    public TreeNode frontSearch(int i) {
        TreeNode tar = null;
        if (this.value == i) return this;
        else {
            if (left != null) tar = left.frontSearch(i);
            if (tar != null) return tar;
            if (right != null) tar = right.frontSearch(i);
        }
        return tar;
    }

    // 删除子树
    public void delete(int i) {
        TreeNode parent = this;

        if (parent.left != null && parent.left.value == i) {
            parent.left = null;
            return;
        }
        if (parent.right != null && parent.right.value == i) {
            parent.right = null;
            return;
        }

        parent = left;
        if (parent != null) parent.delete(i);
        parent = right;
        if (parent != null) parent.delete(i);
    }
}

```

// 线索二叉树
// 左前右后（建树后的空指针域）
public class ThreadedBinaryTree {

    ThreadedNode root;
    // 临时前驱节点
    ThreadedNode pre;

    public ThreadedNode getRoot() {
        return root;
    }

    public void setRoot(ThreadedNode root) {
        this.root = root;
    }

    // 中序线索化二叉树
    public void threadNodes() {
        threadNodes(root);
    }

    public void threadNodes(ThreadedNode node) {
        if (node == null) return;

        // 左子树
        threadNodes(node.left);

        // 前驱节点
        if (node.left == null) {
            node.left = pre;    // 左前
            node.leftType = 1;
        }
        // 前驱的右指针
        if (pre != null && pre.right == null) {
            pre.right = node;   // 右后
            pre.rightType = 1;
        }
        // 每处理一个节点，当前即下一个的前驱
        pre = node;

        // 右子树
        threadNodes(node.right);

    }

    // 遍历线索二叉树(中序)
    public void threadIterate() {
        // 当前
        ThreadedNode node = root;

        while (node != null) {
            // 最开始
            while (node.leftType == 0) {
                node = node.left;
            }
            System.out.println(node.value);

            // 后继
            while (node.rightType == 1) {
                node = node.right;
                System.out.println(node.value);
            }

            // 替换
            node = node.right;
        }
    }


}

```

    public static Node createHuffmanTree(int[] arr) {
        List<Node> nodes = new ArrayList<>();
        Node left, right, parent;

        for (int value : arr) { // convert from array to list
            nodes.add(new Node(value));
        }

        while (nodes.size() > 1) {
            Collections.sort(nodes);    // sort
            left = nodes.get(nodes.size() - 1); // smallest
            right = nodes.get(nodes.size() - 2);
            parent = new Node(left.value + right.value);    // new
            nodes.remove(left); // remove
            nodes.remove(right);
            nodes.add(parent);  // add
        }

        return nodes.get(0);
    }
